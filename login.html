async function login() { let email = document.getElementById("email").value; let password = document.getElementById("password").value; try{ const response = await fetch("http://localhost:5000/api/login",{ method:"POST", headers:{ "Content-Type":"application/json" }, body:JSON.stringify({ email:email, password:password }) }); const data = await response.json(); if(response.ok){ // Save JWT Token localStorage.setItem("token",data.token); // Save User Info localStorage.setItem("user",JSON.stringify(data.user)); // Redirect according to role if(data.user.role==="admin"){ window.location.href="dashboard.html"; }else{ window.location.href="client-dashboard.html"; } }else{ alert(data.message); } }catch(error){ alert("Server not available."); } }
